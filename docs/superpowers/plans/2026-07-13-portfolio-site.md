# Personal Portfolio Site Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers-extended-cc:subagent-driven-development (recommended) or superpowers-extended-cc:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a single-page, dark-only, typography-first portfolio site (react.doctor aesthetic) featuring Capture Studio, LC Finance, and WebSketch2 with inline looping demo videos.

**Architecture:** Next.js App Router site with all content in typed TypeScript data files (`lib/site.ts`, `lib/projects.ts`). One page composed of Header → Projects → About sections. Demo videos are lazy-loading `<video>` elements that autoplay/pause via IntersectionObserver and respect `prefers-reduced-motion`. Videos will eventually live in a public Supabase Storage bucket; placeholder URLs are used until real demos are recorded.

**Tech Stack:** Next.js (App Router), React 19, TypeScript, Tailwind CSS + shadcn/ui (preset `b1au69eka`), Vitest + React Testing Library, npm, Vercel.

**Spec:** `docs/superpowers/specs/2026-07-13-portfolio-site-design.md`

```tsx
"use client";

import { useEffect, useRef, useState } from "react";

interface DemoVideoProps {
  src: string;
  title: string;
}

export function DemoVideo({ src, title }: DemoVideoProps) {
  const videoRef = useRef<HTMLVideoElement>(null);
  const [reducedMotion, setReducedMotion] = useState(false);

  useEffect(() => {
    const mq = window.matchMedia("(prefers-reduced-motion: reduce)");
    setReducedMotion(mq.matches);
    const onChange = (e: MediaQueryListEvent) => setReducedMotion(e.matches);
    mq.addEventListener("change", onChange);
    return () => mq.removeEventListener("change", onChange);
  }, []);

  useEffect(() => {
    if (reducedMotion) return;
    const video = videoRef.current;
    if (!video) return;
    const observer = new IntersectionObserver(
      ([entry]) => {
        if (entry.isIntersecting) {
          video.play().catch(() => {});
        } else {
          video.pause();
        }
      },
      { threshold: 0.25 }
    );
    observer.observe(video);
    return () => observer.disconnect();
  }, [reducedMotion]);

  return (
    <video
      ref={videoRef}
      src={src}
      muted
      loop
      playsInline
      preload="none"
      controls={reducedMotion}
      aria-label={`${title} demo`}
      className="w-full rounded-md border border-border"
    />
  );
}
```
